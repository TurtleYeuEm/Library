                local abs = track.AbsolutePosition
                local sz  = track.AbsoluteSize
                local pct = math.clamp((mx - abs.X) / sz.X, 0, 1)
                val = math.floor(min + pct * (max - min) + 0.5)
                local fpct = (val - min) / (max - min)
                fill.Size = UDim2.new(fpct, 0, 1, 0)
                knob.Position = UDim2.new(fpct, 0, 0.5, 0)
                valLabel.Text = tostring(val)
                if callback then callback(val) end
            end
            knob.InputBegan:Connect(function(inp)
                if inp.UserInputType == Enum.UserInputType.MouseButton1 or inp.UserInputType == Enum.UserInputType.Touch then
                    sliding = true
                end
            end)
            UIS.InputChanged:Connect(function(inp)
                if sliding and (inp.UserInputType == Enum.UserInputType.MouseMovement or inp.UserInputType == Enum.UserInputType.Touch) then
                    updateSlider(inp.Position.X)
                end
            end)
            UIS.InputEnded:Connect(function(inp)
                if inp.UserInputType == Enum.UserInputType.MouseButton1 or inp.UserInputType == Enum.UserInputType.Touch then
                    sliding = false
                end
            end)
            track.InputBegan:Connect(function(inp)
                if inp.UserInputType == Enum.UserInputType.MouseButton1 then
                    updateSlider(inp.Position.X)
                end
            end)
            local ctrl = {}
            function ctrl:Get() return val end
            function ctrl:Set(v)
                val = math.clamp(v, min, max)
                local pct = (val - min) / (max - min)
                fill.Size = UDim2.new(pct, 0, 1, 0)
                knob.Position = UDim2.new(pct, 0, 0.5, 0)
                valLabel.Text = tostring(val)
            end
            return ctrl
        end

        glow.Position = btnPos
        glow.Size = UDim2.new(0, btnSize + 40, 0, btnSize + 40)
        glow.ZIndex = 10
        glow.Image = "rbxassetid://313486536"
        glow.ImageColor3 = Color3.fromRGB(0, 0, 0)
        glow.ImageTransparency = 0.55

        local btn = Instance.new("TextButton")
        btn.Name = "togglebtn"
        btn.Parent = scrgui
        btn.AnchorPoint = Vector2.new(0.5, 0.5)
        btn.Position = btnPos
        btn.Size = UDim2.new(0, btnSize, 0, btnSize)
        btn.AutoButtonColor = false
        btn.Text = ""
        btn.ZIndex = 11
        btn.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
        local uc_tb = Instance.new("UICorner")
        uc_tb.CornerRadius = UDim.new(1, 0)
        uc_tb.Parent = btn

        local grad = Instance.new("UIGradient")
        grad.Parent = btn

        local stroke = Instance.new("UIStroke")
        stroke.Parent = btn
        stroke.Thickness = 2
        stroke.Color = Color3.fromRGB(255, 255, 255)
        stroke.Transparency = 0.6

        local icon = Instance.new("ImageLabel")
        icon.Parent = btn
        icon.AnchorPoint = Vector2.new(0.5, 0.5)
        icon.Position = UDim2.new(0.5, 0, 0.5, 0)
        icon.Size = UDim2.new(0, btnSize * 0.52, 0, btnSize * 0.52)
        icon.BackgroundTransparency = 1
        icon.ScaleType = Enum.ScaleType.Fit
        icon.ZIndex = 12
        icon.Image = iconId

        local function applyTheme(t_name)
            local t = THEMES[t_name] or THEMES.amethyst
            if t.gradient then
                btn.BackgroundTransparency = t.transparency or 0
                btn.BackgroundColor3 = t.gradient[1]
                grad.Color = ColorSequence.new{
                    ColorSequenceKeypoint.new(0, t.gradient[1]),
                    ColorSequenceKeypoint.new(1, t.gradient[2]),
                }
                grad.Rotation = t.rotation or 135
                grad.Enabled = true
                icon.ImageColor3 = Color3.fromRGB(255, 255, 255)
                stroke.Color = Color3.fromRGB(255, 255, 255)
                stroke.Transparency = 0.6
            else
                btn.BackgroundColor3 = t.base
                btn.BackgroundTransparency = t.transparency
                grad.Enabled = false
                if t_name == "light" then
                    icon.ImageColor3 = Color3.fromRGB(30, 30, 30)
                    stroke.Color = Color3.fromRGB(30, 30, 30)
                    stroke.Transparency = 0.7
                else
                    icon.ImageColor3 = Color3.fromRGB(220, 220, 220)
                    stroke.Color = Color3.fromRGB(180, 180, 180)
                    stroke.Transparency = 0.5
                end
            end
        end
        applyTheme(theme)

        local normalSize = UDim2.new(0, btnSize, 0, btnSize)
        local pressSize  = UDim2.new(0, btnSize * 0.88, 0, btnSize * 0.88)

        btn.MouseButton1Down:Connect(function()
            TS:Create(btn,  TweenInfo.new(0.10, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {Size = pressSize}):Play()
            TS:Create(glow, TweenInfo.new(0.10, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {ImageTransparency = 0.75}):Play()
        end)
        btn.MouseButton1Up:Connect(function()
            TS:Create(btn,  TweenInfo.new(0.18, Enum.EasingStyle.Back,  Enum.EasingDirection.Out), {Size = normalSize}):Play()
            TS:Create(glow, TweenInfo.new(0.18, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {ImageTransparency = 0.55}):Play()
        end)
        btn.MouseButton1Click:Connect(function()
            window:ToggleVisible()
            TS:Create(icon, TweenInfo.new(0.35, Enum.EasingStyle.Quart, Enum.EasingDirection.InOut), {Rotation = visible and 0 or 180}):Play()
        end)

        local draggingBtn, btnDragStart, btnStartPos = false, nil, nil
        btn.InputBegan:Connect(function(inp)
            if inp.UserInputType == Enum.UserInputType.MouseButton1 or inp.UserInputType == Enum.UserInputType.Touch then
                draggingBtn = true
                btnDragStart = inp.Position
                btnStartPos  = btn.Position
                inp.Changed:Connect(function()
                    if inp.UserInputState == Enum.UserInputState.End then draggingBtn = false end
                end)
            end
        end)
        UIS.InputChanged:Connect(function(inp)
            if draggingBtn and (inp.UserInputType == Enum.UserInputType.MouseMovement or inp.UserInputType == Enum.UserInputType.Touch) then
                local d = inp.Position - btnDragStart
                local np = UDim2.new(btnStartPos.X.Scale, btnStartPos.X.Offset + d.X, btnStartPos.Y.Scale, btnStartPos.Y.Offset + d.Y)
                btn.Position  = np
                glow.Position = np
            end
        end)

        local ctrl = {}
        function ctrl:SetImage(id) icon.Image = id end
        function ctrl:SetTheme(t) applyTheme(t) end
        function ctrl:SetSize(px)
            normalSize = UDim2.new(0, px, 0, px)
            pressSize  = UDim2.new(0, px * 0.88, 0, px * 0.88)
            btn.Size   = normalSize
            glow.Size  = UDim2.new(0, px + 40, 0, px + 40)
            icon.Size  = UDim2.new(0, px * 0.52, 0, px * 0.52)
        end
        function ctrl:SetPosition(u)
            btn.Position  = u
            glow.Position = u
        end
        return ctrl
    end

    return window
end
